1. Операторы


# Арифметические операторы
print(5 + 3)    # 8
print(10 - 2)   # 8
print(4 * 2)    # 8
print(16 / 2)   # 8.0

# Специальные операторы
print(10 % 3)   # 1 (остаток от деления)
print(10 // 3)  # 3 (целочисленное деление)
print(2 ** 3)   # 8 (возведение в степень)

# Операторы сравнения
print(5 == 5)   # True
print(5 != 3)   # True
print(10 > 5)   # True

# Логические операторы
print(True and False)  # False
print(True or False)   # True
print(not True)        # False

2. Условный оператор


age = int(input("Возраст: "))

if age < 18:
    print("Доступ запрещен")
elif age < 21:
    print("Доступ ограничен")
else:
    print("Полный доступ")

# Тернарный оператор
status = "Adult" if age >= 18 else "Minor"
print(status)

3. Цикл for


# Перебор элементов
fruits = ["яблоко", "банан", "вишня"]
for fruit in fruits:
    print(fruit)

# Использование range()
for i in range(5):
    print(i)  # 0,1,2,3,4

for i in range(2, 10, 2):
    print(i)  # 2,4,6,8

4. Цикл while


count = 0
while count < 5:
    print(count)
    count += 1

# Бесконечный цикл с break
while True:
    command = input("Введите 'exit' для выхода: ")
    if command == 'exit':
        break

7. Словари


user = {
    'name': 'Алексей',
    'age': 25,
    'is_student': True
}

# Доступ к элементам
print(user['name'])

# Добавление/изменение
user['email'] = 'alex@example.com'

# Удаление
del user['is_student']

# Перебор
for key, value in user.items():
